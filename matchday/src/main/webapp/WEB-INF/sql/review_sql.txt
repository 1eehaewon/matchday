-- 리뷰 테이블

CREATE TABLE review (
	reviewid VARCHAR(50) NOT NULL COLLATE,
	userid VARCHAR(20) NOT NULL COLLATE,
	matchid VARCHAR(50) NULL DEFAULT NULL COLLATE,
	goodsid VARCHAR(50) NULL DEFAULT NULL COLLATE,
	reviewdate TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
	title TEXT NOT NULL,
	content TEXT NOT NULL COLLATE,
	rating INT(11) NOT NULL,
	grantedpoints INT(11) NOT NULL,
	PRIMARY KEY (reviewid) USING BTREE,
	INDEX userid (userid) USING BTREE,
	INDEX matchid (matchid) USING BTREE,
	INDEX goodsid (goodsid) USING BTREE,
	CONSTRAINT fk_reviews_goodsid FOREIGN KEY (goodsid) REFERENCES goods (goodsid) ON UPDATE RESTRICT ON DELETE RESTRICT,
	CONSTRAINT fk_reviews_matchid FOREIGN KEY (matchid) REFERENCES matches (matchid) ON UPDATE RESTRICT ON DELETE RESTRICT,
	CONSTRAINT fk_reviews_userid FOREIGN KEY (userid`) REFERENCES users (userid) ON UPDATE RESTRICT ON DELETE RESTRICT
)
